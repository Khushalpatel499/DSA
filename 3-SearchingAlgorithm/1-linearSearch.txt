// linear search:
// time=O(n)
// spaceO(1)

#include <iostream>
#include<vector>
using   namespace std;
int linearSearch(vector<int>&v,int x){
    int index =-1;
    for(int i=0;i<v.size();i++){
        if(v[i]==x){
            index=i;
        }
    }
    return index;
}

int main()
{
    vector<int>v={1, 2, 3, 4};
    int x=3;
    cout<<linearSearch(v,x);

    return 0;
}
